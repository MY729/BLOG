# 闭包

```js
for ( var i=1; i<=5; i++) {
	setTimeout(() => {
		console.log( i );
	}, i*1000 );
}
```

```js
for ( var i=1; i<=5; i++) {
  ((j) => {
    setTimeout(() => {
      console.log( j );
    }, j*1000)
  })(i)
}
```

```js
for ( let i=1; i<=5; i++) {
	setTimeout(() => {
		console.log( i );
	}, i*1000 );
}
```
